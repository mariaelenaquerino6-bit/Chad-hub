local VehicleManager = {
    Players = Players,
    Workspace = Workspace,
    LocalPlayer = LocalPlayer,
    Camera = Workspace.CurrentCamera,
    
    MostrarNotificacao = function(_, msg)
        pcall(function()
            game:GetService("StarterGui"):SetCore("SendNotification", {
                Title = "Aviso",
                Text = msg,
                Duration = 5
            })
        end)
    end,
    
    ToggleFallDamage = function(self, enable)
        if not (self.LocalPlayer.Character and self.LocalPlayer.Character:FindFirstChild("Humanoid")) then
            return false
        end
        
        local humanoid = self.LocalPlayer.Character.Humanoid
        
        if not enable then
            humanoid:SetStateEnabled(Enum.HumanoidStateType.FallingDown, true)
            humanoid:SetStateEnabled(Enum.HumanoidStateType.Ragdoll, true)
            return false
        end
        
        humanoid:SetStateEnabled(Enum.HumanoidStateType.FallingDown, false)
        humanoid:SetStateEnabled(Enum.HumanoidStateType.Ragdoll, false)
        humanoid:SetStateEnabled(Enum.HumanoidStateType.GettingUp, true)
        humanoid.PlatformStand = false
        return true
    end,
    
    TeleportToSeat = function(self, seat, vehicle)
        if not (self.LocalPlayer.Character and self.LocalPlayer.Character:FindFirstChild("Humanoid")) then
            self:MostrarNotificacao("Personagem não encontrado!")
            return false
        end
        
        local humanoid = self.LocalPlayer.Character.Humanoid
        local hrp = self.LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
        
        if not hrp then
            self:MostrarNotificacao("Parte raiz do personagem não encontrada!")
            return false
        end
        
        humanoid.Sit = false
        task.wait(0.1)
        hrp.CFrame = seat.CFrame + Vector3.new(0, 5, 0)
        task.wait(0.1)
        seat:Sit(humanoid)
        task.wait(0.5)
        
        return humanoid.SeatPart == seat
    end,
    
    TeleportToVoid = function(self, vehicle)
        if vehicle then
            if not vehicle.PrimaryPart then
                local primaryPart = vehicle:FindFirstChild("Body", true) or vehicle:FindFirstChild("Chassis", true)
                if not (primaryPart and primaryPart:IsA("BasePart")) then
                    self:MostrarNotificacao("Parte principal do veículo não encontrada!")
                    return
                end
                vehicle.PrimaryPart = primaryPart
            end
            
            local voidPosition = Vector3.new(0, -1000, 0)
            vehicle:SetPrimaryPartCFrame(CFrame.new(voidPosition))
            task.wait(0.5)
        else
            self:MostrarNotificacao("Veículo inválido!")
        end
    end,
    
    TeleportToPlayer = function(self, vehicle, position)
        if vehicle then
            if not vehicle.PrimaryPart then
                local primaryPart = vehicle:FindFirstChild("Body", true) or vehicle:FindFirstChild("Chassis", true)
                if not (primaryPart and primaryPart:IsA("BasePart")) then
                    self:MostrarNotificacao("Parte principal do veículo não encontrada!")
                    return
                end
                vehicle.PrimaryPart = primaryPart
            end
            
            local offset = position + Vector3.new(5, 0, 5)
            vehicle:SetPrimaryPartCFrame(CFrame.new(offset))
            task.wait(0.5)
        else
            self:MostrarNotificacao("Veículo inválido!")
        end
    end,
    
    ExitCarAndReturn = function(self, returnPosition)
        if self.LocalPlayer.Character and self.LocalPlayer.Character:FindFirstChild("Humanoid") then
            local humanoid = self.LocalPlayer.Character.Humanoid
            if humanoid.SeatPart then
                humanoid.Sit = false
            end
            task.wait(0.1)
            if returnPosition then
                self.LocalPlayer.Character:PivotTo(CFrame.new(returnPosition))
            end
        end
    end,
    
    AtualizarListaCarros = function(self)
        local vehiclesFolder = self.Workspace:FindFirstChild("Vehicles")
        local carList = {}
        
        if vehiclesFolder then
            for _, vehicle in ipairs(vehiclesFolder:GetChildren()) do
                if vehicle.Name:match("Car$") then
                    table.insert(carList, vehicle.Name)
                end
            end
        end
        
        return carList
    end
}

TabCar:AddToggle({
    Name = "Matar todos os carros do servidor",
    Description = "Teleporta os carros para o void",
    Default = false,
    Callback = function(enabled)
        local isKillingCars = enabled
        
        if enabled then
            if not (VehicleManager.LocalPlayer.Character and VehicleManager.LocalPlayer.Character:FindFirstChild("HumanoidRootPart")) then
                VehicleManager:MostrarNotificacao("Personagem não encontrado!")
                return
            end
            
            local originalPosition = VehicleManager.LocalPlayer.Character.HumanoidRootPart.Position
            VehicleManager:ToggleFallDamage(true)
            
            spawn(function()
                local vehiclesFolder = VehicleManager.Workspace:FindFirstChild("Vehicles")
                if vehiclesFolder then
                    local allVehicles = {}
                    for _, vehicle in ipairs(vehiclesFolder:GetChildren()) do
                        if vehicle.Name:match("Car$") then
                            table.insert(allVehicles, vehicle)
                        end
                    end
                    
                    for _, vehicle in ipairs(allVehicles) do
                        if not isKillingCars then break end
                        
                        local seat = vehicle:FindFirstChildWhichIsA("VehicleSeat", true)
                        if seat and seat.Occupant == nil and VehicleManager:TeleportToSeat(seat, vehicle) then
                            VehicleManager:TeleportToVoid(vehicle)
                            VehicleManager:ExitCarAndReturn(originalPosition)
                            VehicleManager:MostrarNotificacao("Carro " .. vehicle.Name .. " foi teleportado para o void!")
                            task.wait(1)
                        end
                    end
                    
                    if isKillingCars then
                        isKillingCars = false
                        VehicleManager:ToggleFallDamage(false)
                    end
                else
                    VehicleManager:MostrarNotificacao("Pasta de veículos não encontrada!")
                end
            end)
        else
            VehicleManager:ToggleFallDamage(false)
        end
    end
})

                                    VehicleManager:ExitCarAndReturn(playerPos)
                                else
                                    VehicleManager:MostrarNotificacao("Falha ao sentar no carro!")
                                end
                                
                                VehicleManager:ToggleFallDamage(false)
                            else
                                VehicleManager:MostrarNotificacao("Personagem do jogador não encontrado!")
                            end

local function FlingBus(targetPlayer)
    if targetPlayer and targetPlayer.Character and LocalPlayer.Character then
        local playerChar = LocalPlayer.Character or LocalPlayer.CharacterAdded:Wait()
        local playerHumanoid = playerChar:FindFirstChildOfClass("Humanoid")
        local playerHRP = playerChar:FindFirstChild("HumanoidRootPart")
        
        if playerHumanoid and playerHRP then
            local originalPosition = playerHRP.Position
            
            pcall(function()
                playerHRP.Anchored = true
                playerHRP.CFrame = CFrame.new(Vector3.new(1181.83, 76.08, -1158.83))
                task.wait(0.2)
                playerHRP.Velocity = Vector3.zero
                playerHRP.RotVelocity = Vector3.zero
                playerHRP.Anchored = false
                if playerHumanoid then
                    playerHumanoid:ChangeState(Enum.HumanoidStateType.GettingUp)
                end
            end)
            
            task.wait(0.5)
            
            local args = {"DeleteAllVehicles"}
            pcall(function()
                ReplicatedStorage:WaitForChild("RE"):WaitForChild("1Ca1r"):FireServer(unpack(args))
            end)
            
            args = {"PickingCar", "SchoolBus"}
            pcall(function()
                ReplicatedStorage:WaitForChild("RE"):WaitForChild("1Ca1r"):FireServer(unpack(args))
            end)
            
            task.wait(1)
            
            local vehiclesFolder = WorkspaceService:FindFirstChild("Vehicles")
            if vehiclesFolder then
                local bus = vehiclesFolder:FindFirstChild(LocalPlayer.Name .. "Car")
                if bus then
                    pcall(function()
                        playerHRP.Anchored = true
                        playerHRP.CFrame = CFrame.new(Vector3.new(1171.15, 79.45, -1166.2))
                        task.wait(0.2)
                        playerHRP.Velocity = Vector3.zero
                        playerHRP.RotVelocity = Vector3.zero
                        playerHRP.Anchored = false
                        playerHumanoid:ChangeState(Enum.HumanoidStateType.Seated)
                    end)
                    
                    local startTime = tick()
                    while true do
                        task.wait()
                        if tick() - startTime > 10 then break end
                        
                        if playerHumanoid.Sit then
                            for _, part in ipairs(bus:GetDescendants()) do
                                if part:IsA("BasePart") then
                                    part.CanCollide = false
                                    pcall(function()
                                        part:SetNetworkOwner(nil)
                                    end)
                                end
                            end
                            
                            running = true
                            connection = RunService.Stepped:Connect(function()
                                if running then
                                    for _, part in ipairs(playerChar:GetDescendants()) do
                                        if part:IsA("BasePart") then
                                            part.CanCollide = false
                                        end
                                    end
                                end
                            end)
                            
                            local flingStartTime = tick()
                            local bodyVel = nil
                            
                            flingConnection = RunService.Heartbeat:Connect(function()
                                if not running then return end
                                
                                if not (targetPlayer and targetPlayer.Character) then
                                    running = false
                                    return
                                end
                                
                                local targetHRP = targetPlayer.Character:FindFirstChild("HumanoidRootPart")
                                local targetHumanoid = targetPlayer.Character:FindFirstChild("Humanoid")
                                
                                if not (targetHRP and targetHumanoid) then
                                    running = false
                                    return
                                end
                                
                                if not (playerHRP and playerHumanoid) then
                                    running = false
                                    return
                                end
                                
                                local randomOffset = Vector3.new(math.random(-10, 10), 0, math.random(-10, 10))
                                
                                pcall(function()
                                    local targetPos = targetHRP.Position + randomOffset
                                    bus:PivotTo(CFrame.new(targetPos) * CFrame.Angles(
                                        math.rad(WorkspaceService.DistributedGameTime * 12000),
                                        math.rad(WorkspaceService.DistributedGameTime * 15000),
                                        math.rad(WorkspaceService.DistributedGameTime * 18000)
                                    ))
                                end)
                                
                                local targetSitting = false
                                for _, part in ipairs(bus:GetDescendants()) do
                                    if (part:IsA("Seat") or part:IsA("VehicleSeat")) and part.Name ~= "VehicleSeat" and part.Occupant == targetHumanoid then
                                        targetSitting = true
                                        break
                                    end
                                end
                                
                                if targetSitting or tick() - flingStartTime > 10 then
                                    running = false
                                    flingConnection:Disconnect()
                                    flingConnection = nil
                                    
                                    pcall(function()
                                        bus:PivotTo(CFrame.new(Vector3.new(-59599.73, 2040070.5, -293391.16)))
                                    end)
                                    
                                    bodyVel = Instance.new("BodyVelocity")
                                    bodyVel.MaxForce = Vector3.new(math.huge, math.huge, math.huge)
                                    bodyVel.Velocity = Vector3.new(math.random(-5, 5), 5, math.random(-5, 5)).Unit * 1000000 + Vector3.new(0, 1000000, 0)
                                    bodyVel.Parent = playerHRP
                                    
                                    task.wait(0.5)
                                    
                                    local deleteArgs = {"DeleteAllVehicles"}
                                    pcall(function()
                                        ReplicatedStorage:WaitForChild("RE"):WaitForChild("1Ca1r"):FireServer(unpack(deleteArgs))
                                    end)
                                    
                                    if bodyVel then
                                        bodyVel:Destroy()
                                        for _, part in ipairs(playerChar:GetDescendants()) do
                                            if part:IsA("BasePart") then
                                                part.CanCollide = true
                                            end
                                        end
                                    end
                                    
                                    pcall(function()
                                        playerHRP.Anchored = true
                                        playerHRP.CFrame = CFrame.new(originalPosition + Vector3.new(0, 5, 0))
                                        task.wait(0.2)
                                        playerHRP.Velocity = Vector3.zero
                                        playerHRP.RotVelocity = Vector3.zero
                                        playerHRP.Anchored = false
                                        if playerHumanoid then
                                            playerHumanoid:ChangeState(Enum.HumanoidStateType.GettingUp)
                                        end
                                    end)
                                    
                                    if flingToggle then
                                        flingToggle:Set(false)
                                    end
                                end
                            end)
                            return
                        end
                    end
                end
            else
                return
            end
        else
            return
        end
    else
        return
    end
end

        local targetHRP = targetPlayer.Character:FindFirstChild("HumanoidRootPart")
        if targetHRP then
            local assault = GetAssaultGun()
            local gunScript = assault and assault:FindFirstChild("GunScript_Local")
            local muzzleEffect = gunScript and gunScript:FindFirstChild("MuzzleEffect")
            local hitEffect = gunScript and gunScript:FindFirstChild("HitEffect")
            
            if targetHRP and muzzleEffect and hitEffect then
                local args = {
                    targetHRP,
                    targetHRP,
                    Vector3.new(100000000376832, 100000000376832, 100000000376832),
                    targetHRP.Position,
                    muzzleEffect,
                    hitEffect,
                    0,
                    0,
                    {false},
                    {25, Vector3.new(100, 100, 100), BrickColor.new(29), 0.25, Enum.Material.SmoothPlastic, 0.25},
                    true,
                    false
                }
                
                pcall(function()
                    local RE_Gun = ReplicatedStorage:FindFirstChild("RE") and ReplicatedStorage.RE:FindFirstChild("1Gu1n")
                    if RE_Gun then
                        RE_Gun:FireServer(unpack(args))
                    end
                end)
            end
        end
    else
        return
    end
end

local function SetupVehicleCollisions()
    for _, vehicle in pairs(Workspace:WaitForChild("Vehicles"):GetChildren()) do
        local vehicleTypes = {
            "Shark", "WaveRunner", "SailBoatFree", "SpeedBoatFree",
            "PoliceBoatChangeable", "MilitaryBoatFree", "PirateFree",
            "Jet", "Helicopter"
        }
        
        for _, part in pairs(vehicle:GetDescendants()) do
            if part:IsA("BasePart") then
                local partName = part.Name:lower()
                local isWaterVehicle = false
                
                for _, vType in ipairs(vehicleTypes) do
                    if vehicle.Name == vType then
                        isWaterVehicle = true
                        break
                    end
                end
                
                part.CanCollide = isWaterVehicle and (
                    part:IsA("Seat") or part:IsA("VehicleSeat") or
                    partName:find("wheel") or partName:find("tire") or
                    partName:find("chassis") or partName:find("base") or
                    partName:find("frame") or partName:find("hull") or
                    partName:find("propeller") or partName:find("rotor") or
                    partName:find("wing")
                ) or (
                    part:IsA("Seat") or part:IsA("VehicleSeat") or
                    partName:find("wheel") or partName:find("tire") or
                    partName:find("chassis") or partName:find("base") or
                    partName:find("frame")
                )
            end
        end
    end
end

                end)
            end
            
            wait(0.3)
        end)
    else
        if laptopLagConnection then
            laptopLagConnection:Disconnect()

        end

local TabAntiLag = Window:MakeTab({"Anti Lag Server", "alert"})

TabAntiLag:AddSection({"Proteção contra lag server de outros scripts"})

local antiLagLimits = {
    Laptop = false,
    FireX = false,
    Bomb = false,
    Stretcher = false
}

local antiLagConnections = {}

local function SetupAntiLagForPlayer(player, itemName)
    local function MonitorContainer(container)
        container.ChildAdded:Connect(function(child)
            if antiLagLimits[itemName] and child.Name == itemName then
                local count = 0
                for _, item in ipairs(container:GetChildren()) do
                    if item.Name == itemName then
                        count = count + 1
                        if count > 10 then
                            item.Parent = nil
                        end
                    end
                end
            end
        end)
    end
    
    local backpack = player:FindFirstChild("Backpack")
    if backpack then
        MonitorContainer(backpack)
    end
    
    if player.Character then
        MonitorContainer(player.Character)
    end
    
    player.CharacterAdded:Connect(function(char)
        MonitorContainer(char)
    end)
end

local function EnableAntiLag(itemName)
    for _, player in ipairs(Players:GetPlayers()) do
        SetupAntiLagForPlayer(player, itemName)
    end
    
    antiLagConnections[itemName] = Players.PlayerAdded:Connect(function(player)
        SetupAntiLagForPlayer(player, itemName)
    end)
end

local function DisableAntiLag(itemName)
    if antiLagConnections[itemName] then
        antiLagConnections[itemName]:Disconnect()
        antiLagConnections[itemName] = nil
    end
end

local function CreateAntiLagToggle(name, itemKey)
    TabAntiLag:AddToggle({
        Name = name,
        Flag = "AntiLag_" .. itemKey,
        Default = false,
        Callback = function(enabled)
            antiLagLimits[itemKey] = enabled
            if enabled then
                EnableAntiLag(itemKey)
                vu1:Notify("Anti Lag Server", name .. " limitado a 10 visíveis!")
            else
                DisableAntiLag(itemKey)
                vu1:Notify("Anti Lag Server", name .. " sem limite!")
            end
        end
    })
end

local RGBSettings = {
    Speed = 3,
    States = {}
}

local function GetRGBColor()
    local hue = tick() * RGBSettings.Speed % 5 / 5
    return Color3.fromHSV(hue, 1, 1)
end

local function StartRGBEffect(config)
    RGBSettings.States[config.StateKey] = true
    
    task.spawn(function()
        while RGBSettings.States[config.StateKey] do
            if not (LocalPlayer and LocalPlayer.Parent) then
                RGBSettings.States[config.StateKey] = false
                break
            end
            config.Action()
            task.wait(config.Delay)
        end
    end)
end

local function CreateRGBToggle(tab, config)
    tab:AddToggle({
        Name = config.Name,
        Description = config.Description,
        Default = false,
        Callback = function(enabled)
            if enabled then
                StartRGBEffect(config)
            else
                RGBSettings.States[config.StateKey] = false
            end
        end
    })
end
